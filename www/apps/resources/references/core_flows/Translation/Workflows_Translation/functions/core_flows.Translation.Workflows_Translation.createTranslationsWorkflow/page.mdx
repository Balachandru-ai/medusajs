# createTranslationsWorkflow - Medusa Core Workflows Reference

This documentation provides a reference to the `createTranslationsWorkflow`. It belongs to the `@medusajs/medusa/core-flows` package.

This workflow creates one or more translations.

You can use this workflow within your own customizations or custom workflows, allowing you
to create translations in your custom flows.

<SourceCodeLink link="https://github.com/medusajs/medusa/blob/75eac9ebec217fd299c6e5d8f161a597b46e4e19/packages/core/core-flows/src/translation/workflows/create-translations.ts#L42" />

    <CodeTab label="API Route" value="api-route">
    
```ts title="src/api/workflow/route.ts"
import type {
  MedusaRequest,
  MedusaResponse,
} from "@medusajs/framework/http"
import { createTranslationsWorkflow } from "@medusajs/medusa/core-flows"

export async function POST(
  req: MedusaRequest,
  res: MedusaResponse
) {
  const { result } = await createTranslationsWorkflow(req.scope)
    .run({
      input: {
        translations: [{
          reference_id: "prod_123",
          reference: "product",
          locale_code: "fr-FR",
          translations: { title: "Produit",
            description: "Description du produit" }
        }]
      }
    })

  res.send(result)
}
```
