# updateTranslationsWorkflow - Medusa Core Workflows Reference

This documentation provides a reference to the `updateTranslationsWorkflow`. It belongs to the `@medusajs/medusa/core-flows` package.

This workflow updates translations matching the specified filters.

You can use this workflow within your own customizations or custom workflows, allowing you
to update translations in your custom flows.

<SourceCodeLink link="https://github.com/medusajs/medusa/blob/75eac9ebec217fd299c6e5d8f161a597b46e4e19/packages/core/core-flows/src/translation/workflows/update-translations.ts#L39" />

```ts title="src/api/workflow/route.ts"
import type {
  MedusaRequest,
  MedusaResponse,
} from "@medusajs/framework/http"
import { updateTranslationsWorkflow } from "@medusajs/medusa/core-flows"

export async function POST(
  req: MedusaRequest,
  res: MedusaResponse
) {
  const { result } = await updateTranslationsWorkflow(req.scope)
    .run({
      input: {
        selector: {
          reference_id: "prod_123",
          locale_code: "fr-FR"
        },
        update: {
          translations: { title: "Nouveau titre" }
        }
      }
    })

  res.send(result)
}
```
